可以指定编译使用的jdk版本等
<bulid>标签和<dependencies>标签同级
<bulid>
	<plugins>
		<plugin>
			<groupId>org.apache.maven.plugins</groupId>
			<artifactId>maven-compiler-plugin</artifactId>
			<configuration>
				<source>1.8</source>
				<target>1.8</target>
				<encodiong>UTF-8</encodiong>
			</configuration>
		</plugin>
	</plugins>
</bulid>
